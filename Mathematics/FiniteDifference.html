evalD <- function(f, x) {
    h <- 0.001
    return((f(x + h) - f(x - h))/(2 * h))
}

fprime <- evalD(function(x) {
    x^3 + 2 * x^2 + 3
}, 0:10)

Df <- function(x) {
    return(3 * x^2 + 4 * x)

evalDh <- function(f, x, h) {
    # Adjusting EvalD to take h has an argument
    return((f(x + h) - f(x - h))/(2 * h))
}
