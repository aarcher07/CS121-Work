DrawCircle <- function(x, y, r, ...) {
    angles <- seq(from = 0, to = 2 * pi, length = 1000)
    xpts <- x + r * cos(angles)
    ypts <- y + r * sin(angles)
    polygon(xpts, ypts, asp = 1, ...)
}

DrawCircle(17, 47, 10, border = "blue")
DrawCircle(30, 38, 10, border = "yellow")
